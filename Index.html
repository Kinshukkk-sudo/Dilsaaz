  <!-- Botpress Chat Embed -->

  <script src="https://cdn.botpress.cloud/webchat/v0/inject.js"></script>

  <script>

    window.botpressWebChat.init({

      configUrl: "https://files.bpcontent.cloud/2025/06/22/07/20250622075441-786C3QEB.json",

      botName: "Dilsaaz",

      showPoweredBy: false,

      enableTranscriptDownload: false,

      containerWidth: "100%",

      layoutWidth: "100%"

    });

  </script>


  <!-- Voice Input + Reply Script -->

  <script>

    const micButton = document.getElementById('mic');

    const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();

    recognition.lang = 'en-IN';


    micButton.addEventListener('click', () => {

      recognition.start();

    });


    recognition.onresult = (event) => {

      const userSpeech = event.results[0][0].transcript;

      window.botpressWebChat.sendEvent({ type: 'proactive-trigger', channel: 'web', payload: { text: userSpeech } });

    };


    // Make Dilsaaz speak her replies

    window.botpressWebChat.onEvent(async function (event) {

      if (event.type === 'message' && event.payload && event.payload.text) {

        const utter = new SpeechSynthesisUtterance(event.payload.text);

        utter.lang = 'en-IN';

        utter.rate = 0.95;

        utter.pitch = 1;

        speechSynthesis.speak(utter);

      }

    });

  </script>
